### 문제 링크
* https://icpc.me/12930

### 사용 알고리즘
* Dijkstra

### 풀이
다익스트라 알고리즘을 쓰면 쉽게 풀 수 있는 문제입니다.<br>
priority_queue에 정점과 거리 외에도 추가로 W1의 합과 W2의 합도 넣어야 합니다.

### 전체 코드
```cpp
#include <bits/stdc++.h>
#define x first
#define y second
using namespace std;

typedef long long ll;
typedef pair<ll, ll> p;
typedef pair<ll, p> edge;
typedef pair<p, p> pp;

char arr[22][22][2];
vector<edge> g[22];
ll dst[22];

int n;

int main(){
    ios_base::sync_with_stdio(0); cin.tie(0);
    cin >> n;
    for(int k=0; k<2; k++){
        for(int i=1; i<=n; i++){
            for(int j=1; j<=n; j++){
                cin >> arr[i][j][k];
            }
        }
    }
    for(int i=1; i<=n; i++){
        for(int j=1; j<=n; j++){
            if(arr[i][j][0] == '.') continue;
            g[i].emplace_back((ll)j, (p){arr[i][j][0] - '0', arr[i][j][1] - '0'});
        }
    }
    memset(dst, 0x3f, sizeof dst);
    priority_queue<pp> pq; pq.push({{0, 1}, {0, 0}}); dst[1] = 0;
    while(pq.size()){
        ll cst = -pq.top().x.x;
        ll now = pq.top().x.y;
        ll a = pq.top().y.x;
        ll b = pq.top().y.y;
        pq.pop();
        if(cst > dst[now]) continue;
        for(auto i : g[now]){
            ll nxt = i.x;
            ll aa = i.y.x, bb = i.y.y;
            ll nxtCst = (a + aa) * (b + bb);
            if(nxtCst < dst[nxt]){
                dst[nxt] = nxtCst;
                pq.push({{-nxtCst, nxt}, {a+aa, b+bb}});
            }
        }
    }
    if(dst[2] > 1e18) cout << -1;
    else cout << dst[2];
}
```
